cstring filename = System.Constants.DataDirectory + "/LkVat2.csv";

int fileHandle = OpenFile(filename, 0);

ReadFileLine(fileHandle); // ignore header




int processIndex;
cstring line;
for (processIndex = 0; processIndex != System.Constants.LeakTestCount; ++processIndex)
{

	line = ReadFileLine(fileHandle);
	if (line == "") break; // No more data, so we're done
	
	Utility.Tokenize(line, ',');
	
	// The order of data items must match those in SaveProcesses

	Leak_Test.DEP.Vat2.HMI.resultsLog.date[processIndex] = Utility.NextToken();
	Leak_Test.DEP.Vat2.HMI.resultsLog.staticTolerance[processIndex] = TextToFloat(Utility.NextToken());
	Leak_Test.DEP.Vat2.HMI.resultsLog.dynamicTolerance[processIndex] = TextToFloat(Utility.NextToken());
	Leak_Test.DEP.Vat2.HMI.resultsLog.beginningPressure[processIndex] = TextToFloat(Utility.NextToken());
	Leak_Test.DEP.Vat2.HMI.resultsLog.sampledPressure[processIndex] = TextToFloat(Utility.NextToken());
	Leak_Test.DEP.Vat2.HMI.resultsLog.greatestDelta[processIndex] = TextToFloat(Utility.NextToken());
	Leak_Test.DEP.Vat2.HMI.resultsLog.result[processIndex] = Utility.NextToken();
}


CloseFile(fileHandle);
